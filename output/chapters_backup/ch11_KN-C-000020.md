Python

import time

# Tool 1：模拟天气API
def weather_api(city):
    print("调用天气服务...")
    time.sleep(1)

    mock_response = {
        "北京": {"weather": "晴天"},
        "上海": {"weather": "小雨"},
        "深圳": {"weather": "多云"}
    }

    return mock_response.get(
        city,
        {"weather": "未知"}
    )

# Tool 2：建议生成工具
def generate_advice(weather):
    rules = {
        "晴天": "短袖即可",
        "小雨": "带雨伞",
        "多云": "薄外套"
    }

    return rules.get(
        weather,
        "查看实时天气"
    )

# Agent
def weather_agent(city):

    print(f"\n任务：查询{city}天气\n")

    # Thought
    print("Thought: 我需要先查询天气")

    # Action
    result = weather_api(city)

    print(f"Action: weather_api('{city}')")

    # Observation
    weather = result["weather"]

    print(f"Observation: {weather}")

    print("Thought: 根据天气给出建议")

    # Action
    advice = generate_advice(weather)

    print(
        f"Action: generate_advice('{weather}')"
    )

    # Observation
    print(f"Observation: {advice}")

    print("\n最终结果")

    print(f"{city}天气：{weather}")
    print(f"建议：{advice}")

if __name__ == "__main__":
    weather_agent("上海")
