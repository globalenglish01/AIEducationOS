## 什么时候该Fine-tuning

张博在白板上画了一张决策树。

```
你的问题是什么？
│
├── 需要检索外部文档？
│   └── 用 RAG（不需要训练，动态更新）
│
├── 知识能放进Context（几千字以内）？
│   └── 用 Prompt Engineering（快速验证，零成本）
│
├── 需要学习特定风格/格式？
│   └── 用 Fine-tuning（让模型学会"怎么做"）
│
├── 需要在特定任务上改变行为？
│   └── 用 Fine-tuning（让模型"忘掉"旧习惯）
│
└── 需要完全不同的能力（如代码执行）？
    └── 考虑工具调用 + Agent
```

"合同工作日的问题，"张博说，"不是知识问题，是**行为**问题。模型知道国家工作日是什么，它需要学的是：在审核你们公司合同时，要用你们公司的标准，而不是国家标准。这是行为的改变，不是知识的增加。"

"所以RAG没用？"

"RAG能告诉它工作日政策，但改变不了它的判断习惯。Fine-tuning能做到：每次遇到工作日相关的合同条款，自动用公司标准来判断，不需要你每次提醒它。"

陈浩明白了。

**Fine-tuning的三种目标：**

1. **风格学习**：让模型学会用特定的语气、格式输出（比如：用公司统一的审核报告模板）
2. **领域知识内化**：让模型把公司特有规则变成"直觉"，而不是每次靠提示
3. **任务特化**：让模型在特定任务上表现比通用模型好得多

合同审核属于第2和第3类的结合。

陈浩去和林晓丽确认之后，回来脸色有点难看。

五百份里：

- 约65份标注了"待确认"但最终还是过了

```python
import json
from pathlib import Path

def prepare_training_data(
    raw_records: list,
    output_path: Path,
    quality_threshold: float = 0.85,
) -> tuple[int, int]:
    """
    将人工标注的合同审核记录转换为OpenAI Fine-tuning格式。
    过滤低质量样本，返回（总记录数，有效记录数）。
    """
    valid_count = 0

    with output_path.open("w", encoding="utf-8") as f:
        for record in raw_records:
            # 过滤低质量样本
            if record["annotation_confidence"] < quality_threshold:
                continue
            if record.get("has_disagreement", False):
                continue

            training_example = {
                "messages": [
                    {
                        "role": "system",
                        "content": (
                            "你是XX公司专属合同审核AI。\n"
                            "你深度理解XX公司的合同规范、内部工作日定义和违约条款标准。\n"
                            "审核时，XX公司工作日定义包含：法定假日 + 公司延假（如春节延假）。\n"
                            "三十个工作日 = 约七周（非国家标准的六周）。"
                        ),
                    },
                    {
                        "role": "user",
                        "content": f"请审核以下合同条款：\n\n{record['contract_clause']}",
                    },
                    {
                        "role": "assistant",
                        "content": record["expert_review"],
                    },
                ]
            }

            f.write(json.dumps(training_example, ensure_ascii=False) + "\n")
            valid_count += 1

    return len(raw_records), valid_count


def validate_training_file(file_path: Path) -> dict:
    """验证训练文件格式，统计token数量和估算费用"""
    examples = []
    total_tokens = 0

    with file_path.open(encoding="utf-8") as f:
        for i, line in enumerate(f):
            try:
                example = json.loads(line)
                examples.append(example)
                content = json.dumps(example, ensure_ascii=False)
                total_tokens += len(content) // 4  # 粗估
            except json.JSONDecodeError as e:
                return {"valid": False, "error": f"Line {i+1}: {e}"}

    return {
        "valid": True,
        "example_count": len(examples),
        "estimated_tokens": total_tokens,
        "estimated_cost_usd": round(total_tokens * 0.000008, 4),
    }
```

训练文件准备好后，陈浩运行了验证：

```

estimated_tokens: 284,160
estimated_cost_usd: 2.27
```

## OpenAI Fine-tuning API实战

```python
import asyncio
import time
from openai import AsyncOpenAI
from pathlib import Path

client = AsyncOpenAI()


async def run_finetuning_job(
    training_file: Path,
    validation_file: Path | None = None,
    base_model: str = "gpt-4o-mini-2024-07-18",
    suffix: str = "contract-review",
    n_epochs: int = 3,
    learning_rate_multiplier: float = 1.0,
) -> dict:
    """提交Fine-tuning任务，等待完成，返回模型信息"""
    # 上传训练文件
    print("Uploading training file...")
    with training_file.open("rb") as f:
        train_response = await client.files.create(file=f, purpose="fine-tune")

    val_file_id = None
    if validation_file:
        with validation_file.open("rb") as f:
            val_response = await client.files.create(file=f, purpose="fine-tune")
        val_file_id = val_response.id

    # 提交Fine-tuning任务
    print(f"Starting fine-tuning job (base: {base_model})...")
    job = await client.fine_tuning.jobs.create(
        training_file=train_response.id,
        validation_file=val_file_id,
        model=base_model,
        suffix=suffix,
        hyperparameters={
            "n_epochs": n_epochs,
            "learning_rate_multiplier": learning_rate_multiplier,
            "batch_size": "auto",
        },
    )

    print(f"Job created: {job.id}")

    # 轮询等待完成
    while job.status not in ("succeeded", "failed", "cancelled"):
        await asyncio.sleep(60)
        job = await client.fine_tuning.jobs.retrieve(job.id)

        events = await client.fine_tuning.jobs.list_events(job.id, limit=5)
        for event in reversed(events.data):
            print(f"  [{event.created_at}] {event.message}")

    if job.status != "succeeded":
        raise RuntimeError(f"Job {job.id} failed: {job.error}")

    print(f"Fine-tuning complete! Model: {job.fine_tuned_model}")
    return {
        "job_id": job.id,
        "model_id": job.fine_tuned_model,
        "trained_tokens": job.trained_tokens,
        "training_cost_usd": round(job.trained_tokens * 0.000008, 4),
    }
```

第一次训练跑完，陈浩兴奋地去测试——结果发现训练集表现完美，验证集准确率只有71%，比原来的通用模型还差。

"过拟合了，"张博看了一眼训练曲线，"n_epochs=3，模型把训练集背下来了，泛化能力很差。"

解决方案：把`n_epochs`从3降到2，`learning_rate_multiplier`从1.0降到0.8，同时确保验证集包含了训练集中没见过的合同类型。

第二轮跑完：验证集准确率89%。

"还能再推吗？"

"别贪，"李萍说，"89%和94%之间，可能只差几十条高质量数据，不差超参数。把数据质量再提高，而不是继续调参。"

他们又从法务部收集了30份高质量案例，重新训练了一次。最终验证集准确率：**94%**。

## LoRA：低成本的本地微调

Fine-tuning OpenAI模型很方便，但有一个问题：你的训练数据需要上传给OpenAI。

对于法律合同，这涉及数据隐私。林晓丽后来提了这个顾虑："我们的合同内容有些是商业机密，能不能不传出去？"

这就是本地Fine-tuning的场景。用LoRA（Low-Rank Adaptation）技术，可以在本地GPU上微调开源模型，成本大幅降低，数据不出内网：

```python
# 需要安装：pip install unsloth trl transformers datasets
from unsloth import FastLanguageModel
from trl import SFTTrainer
from transformers import TrainingArguments
from datasets import Dataset


def finetune_local_model(
    training_data: list[dict],
    base_model: str = "Qwen/Qwen2.5-7B-Instruct",
    output_dir: str = "./contract-review-model",
    lora_rank: int = 16,
) -> str:
    """
    用QLoRA（4bit量化 + LoRA）微调本地模型。
    只训练极少数参数（约0.1%），显存需求大幅降低。
    """
    # 加载基础模型（4bit量化，大幅降低显存需求）
    model, tokenizer = FastLanguageModel.from_pretrained(
        model_name=base_model,
        max_seq_length=2048,
        dtype=None,
        load_in_4bit=True,  # QLoRA的关键：4bit量化
    )

    # 添加LoRA适配器
    # 只训练注意力层的Query和Value投影矩阵（约0.1%参数）
    model = FastLanguageModel.get_peft_model(
        model,
        r=lora_rank,
        target_modules=["q_proj", "v_proj", "k_proj", "o_proj"],
        lora_alpha=lora_rank * 2,
        lora_dropout=0.05,
        bias="none",
        use_gradient_checkpointing="unsloth",
    )

    # 准备数据集
    dataset = Dataset.from_list([
        {
            "text": tokenizer.apply_chat_template(
                example["messages"],
                tokenize=False,
                add_generation_prompt=False,
            )
        }
        for example in training_data
    ])

    trainer = SFTTrainer(
        model=model,
        tokenizer=tokenizer,
        train_dataset=dataset,
        dataset_text_field="text",
        max_seq_length=2048,
        args=TrainingArguments(
            output_dir=output_dir,
            num_train_epochs=3,
            per_device_train_batch_size=4,
            gradient_accumulation_steps=4,
            learning_rate=2e-4,
            lr_scheduler_type="cosine",
            warmup_ratio=0.1,
            save_strategy="epoch",
            logging_steps=10,
        ),
    )

    trainer.train()

    # 保存合并后的模型
    model.save_pretrained_merged(
        f"{output_dir}/merged",
        tokenizer,
        save_method="merged_16bit",
    )

    return f"{output_dir}/merged"
```

LoRA的核心思想：**不改变原始模型的全部权重，只在关键层上添加低秩矩阵**。就像在书页上贴便利贴，而不是重印整本书。

和OpenAI Fine-tuning对比：

| 维度 | OpenAI Fine-tuning | LoRA本地微调 |
|------|-------------------|-------------|
| 数据隐私 | 需要上传OpenAI | 数据不出本地 |
| 硬件需求 | 无（云端） | GPU（24GB以上） |
| 训练成本 | 按token付费（便宜） | 电费+GPU折旧 |
| 模型质量 | GPT-4o-mini为基础 | 取决于选择的基础模型 |
| 部署灵活性 | 受OpenAI服务限制 | 完全自主部署 |

对于有数据安全要求的场景，LoRA是更好的选择。

## Fine-tuning的陷阱

用了半年之后，陈浩总结了几个真实踩过的坑：

**坑1：灾难性遗忘**

Fine-tuning后，合同审核准确率提升了，但有一天林晓丽发来消息："模型现在连普通的中文问答都变差了，我问它一些基础法律常识，它给出了奇怪的回答。"

这是**灾难性遗忘**——模型在学习新任务时，覆盖了原来的通用能力。

解决方案：用更小的学习率，减少epoch；或者保留一部分通用问答数据在训练集中，防止能力衰退。

**坑2：训练分布和生产分布不匹配**

Fine-tuned模型在测试集上表现很好，但上线后某类新型合同（劳务外包合同）的准确率很低。调查发现，训练数据里几乎没有这类合同。

**Fine-tuning只能学会它见过的**。对于训练集覆盖不到的领域，它和通用模型一样迷茫。

**坑3：过度依赖Fine-tuning**

某次产品升级，合同格式改了，Fine-tuned模型突然准确率大跌。原来它把格式特征和判断逻辑耦合在一起了。

**Fine-tuning应该训练逻辑，而不是格式**。格式变化要靠Prompt来处理。

## 持续Fine-tuning：数据飞轮

Fine-tuning不是一次性的，而是一个持续迭代的过程：

```python
from datetime import datetime, timedelta


class ContinuousFineTuner:
    """
    持续Fine-tuning系统：数据飞轮的实现。
    用户反馈 → 积累高质量样本 → 定期重训 → 更好的模型 → 更多用户使用
    """

    def __init__(self, db, current_model_id: str):
        self.db = db
        self.current_model_id = current_model_id

    async def collect_feedback(
        self,
        contract_clause: str,
        ai_response: str,
        user_correction: str | None,
        user_rating: int,  # 1-5
    ) -> None:
        """收集律师对AI审核结果的反馈"""
        if user_rating >= 4:
            # 律师认可的结果：作为正样本
            await self.db.insert_candidate(
                clause=contract_clause,
                response=ai_response,
                quality_score=user_rating / 5.0,
                source="lawyer_approval",
            )

        elif user_correction:
            # 律师纠正：最有价值的训练数据
            await self.db.insert_candidate(
                clause=contract_clause,
                response=user_correction,  # 用纠正后的答案
                quality_score=0.95,
                source="lawyer_correction",
            )

    async def should_retrain(self) -> bool:
        last_training = await self.db.get_last_training_time()
        since_last = datetime.utcnow() - last_training
        new_samples = await self.db.count_new_candidates(since=last_training)

        # 每4周且至少100条新样本，或积累超过500条
        return (since_last > timedelta(weeks=4) and new_samples >= 100) or new_samples >= 500

    async def run_incremental_training(self) -> str:
        """增量训练：在上一版模型基础上继续"""
        last_training = await self.db.get_last_training_time()
        new_data = await self.db.get_high_quality_candidates(
            min_quality=0.85,
            since=last_training,
        )

        training_file = self._prepare_file(new_data)
        result = await run_finetuning_job(
            training_file=training_file,
            base_model=self.current_model_id,  # 从当前模型继续
            n_epochs=1,  # 增量训练只需1个epoch
        )

        # A/B测试新旧模型，确认效果后再切换
        print(f"New model ready: {result['model_id']}")
        print("Starting A/B test (10% traffic)...")
        return result["model_id"]
```

这就是**数据飞轮**的核心逻辑：
1. 模型上线 → 律师使用
2. 律师纠正错误 → 高质量数据积累
3. 定期重训 → 更好的模型
4. 更好的模型 → 律师更信任，使用更多
5. 更多使用 → 更多纠正数据
6. 循环

六个月后，模型已经迭代了三次。每次迭代，准确率又提高几个百分点。最新版本的准确率：97.2%。
