"那怎么做？"陈浩问。

"用向量，"AI导师说，"把问题转成Embedding，用余弦相似度判断两个问题是不是在说同一件事。这叫语义缓存（Semantic Cache）。这就是不让模型重复猜相同的事。"

陈浩理解了核心思路，但实现细节还没想清楚：阈值怎么定？缓存满了怎么淘汰？TTL怎么设？他打开VSCode，开始认真实现：

```python
# semantic_cache.py
import hashlib
import numpy as np
from datetime import datetime, timedelta
from dataclasses import dataclass, field
from typing import Optional
import asyncio


@dataclass
class CacheEntry:
    """语义缓存的一条记录"""
    query: str                          # 原始查询
    embedding: list[float]              # 查询的向量表示
    answer: str                         # 缓存的答案
    timestamp: datetime                 # 写入时间
    hit_count: int = 0                  # 命中次数（用于LRU统计）
    last_hit: datetime = field(default_factory=datetime.utcnow)  # 最后命中时间


def cosine_similarity(a: list[float], b: list[float]) -> float:
    """计算余弦相似度，范围 [-1, 1]，越接近1越相似"""
    vec_a = np.array(a)
    vec_b = np.array(b)
    norm_a = np.linalg.norm(vec_a)
    norm_b = np.linalg.norm(vec_b)
    if norm_a == 0 or norm_b == 0:
        return 0.0
    return float(np.dot(vec_a, vec_b) / (norm_a * norm_b))


class SemanticCache:
    """
    基于向量相似度的语义缓存。

    与传统字符串缓存的核心区别：
    - 字符串缓存：完全匹配才命中（精确相等）
    - 语义缓存：向量相似度 >= threshold 即命中（语义相近）

    例：
    "差旅报销怎么走流程" 和 "出差费用怎么申请报销"
    → 字符串缓存：Miss（字符不同）
    → 语义缓存：Hit（相似度约0.94，超过阈值0.92）
    """

    def __init__(
        self,
        similarity_threshold: float = 0.92,
        max_size: int = 1000,
        ttl_hours: int = 24,
    ):
        self.cache: list[CacheEntry] = []
        self.threshold = similarity_threshold
        self.max_size = max_size
        self.ttl = timedelta(hours=ttl_hours)
        self._lock = asyncio.Lock()

        # 监控指标
        self._total_requests = 0
        self._cache_hits = 0

    async def get(
        self, query: str, query_embedding: list[float]
    ) -> Optional[str]:
        """
        查语义缓存。

        算法：
        1. 过滤已过期条目
        2. 对所有有效条目计算余弦相似度
        3. 找到最高相似度条目，若 >= threshold，返回答案

        时间复杂度：O(n)，n为缓存条目数
        """
        async with self._lock:
            self._total_requests += 1
            now = datetime.utcnow()

            best_similarity = 0.0
            best_entry: Optional[CacheEntry] = None

            for entry in self.cache:
                # 跳过过期条目
                if now - entry.timestamp > self.ttl:
                    continue

                similarity = cosine_similarity(query_embedding, entry.embedding)
                if similarity > best_similarity:
                    best_similarity = similarity
                    best_entry = entry

            if best_entry and best_similarity >= self.threshold:
                # 命中，更新访问统计
                best_entry.hit_count += 1
                best_entry.last_hit = now
                self._cache_hits += 1
                return best_entry.answer

            return None

    async def set(
        self, query: str, embedding: list[float], answer: str
    ) -> None:
        """
        写入缓存。若超出容量，淘汰最少使用的条目（LRU策略）。
        """
        async with self._lock:
            # 清理过期条目
            now = datetime.utcnow()
            self.cache = [
                e for e in self.cache
                if now - e.timestamp <= self.ttl
            ]

            # 若仍然超出容量，淘汰最少使用的
            if len(self.cache) >= self.max_size:
                # 按 hit_count 升序排，淘汰最少命中的
                self.cache.sort(key=lambda e: (e.hit_count, e.last_hit))
                self.cache = self.cache[self.max_size // 4:]  # 淘汰25%

            self.cache.append(CacheEntry(
                query=query,
                embedding=embedding,
                answer=answer,
                timestamp=now,
            ))

    @property
    def hit_rate(self) -> float:
        """缓存命中率"""
        if self._total_requests == 0:
            return 0.0
        return self._cache_hits / self._total_requests

    def stats(self) -> dict:
        """返回当前缓存统计信息"""
        return {
            "total_requests": self._total_requests,
            "cache_hits": self._cache_hits,
            "hit_rate": f"{self.hit_rate:.1%}",
            "cache_size": len(self.cache),
            "max_size": self.max_size,
            "threshold": self.threshold,
        }
```

```python
# rag_pipeline.py（集成语义缓存版）
class RAGPipeline:
    def __init__(self):
        self.semantic_cache = SemanticCache(
            similarity_threshold=0.92,  # 经验值：低了会把不同问题当同一个，高了命中率太低
            max_size=1000,
            ttl_hours=24,
        )
        self.embedder = OpenAIEmbedder()
        self.retriever = QdrantRetriever()
        self.llm = OpenAIChat()

    async def query(self, user_query: str) -> dict:
        start_time = time.time()

        # Step 1：对查询做Embedding（无论如何都要做，用于缓存查找）
        query_embedding = await self.embedder.embed(user_query)
        embed_time = time.time() - start_time

        # Step 2：先查语义缓存
        cached_answer = await self.semantic_cache.get(user_query, query_embedding)
        if cached_answer:
            return {
                "answer": cached_answer,
                "cache_hit": True,
                "total_latency_ms": (time.time() - start_time) * 1000,
                "embed_latency_ms": embed_time * 1000,
            }

        # Step 3：缓存Miss，走完整RAG流程
        t1 = time.time()
        docs = await self.retriever.search(query_embedding, top_k=10)
        retrieve_time = time.time() - t1

        t2 = time.time()
        answer = await self.llm.generate(user_query, docs)
        llm_time = time.time() - t2

        # Step 4：写入语义缓存
        await self.semantic_cache.set(user_query, query_embedding, answer)

        return {
            "answer": answer,
            "cache_hit": False,
            "total_latency_ms": (time.time() - start_time) * 1000,
            "embed_latency_ms": embed_time * 1000,
            "retrieve_latency_ms": retrieve_time * 1000,
            "llm_latency_ms": llm_time * 1000,
        }
```

阈值设0.92，这个数字是AI导师给的经验值。陈浩问过为什么，AI导师说："0.95以上，太严格，命中率个位数；0.85以下，太宽松，容易把不同问题当一个。0.92是大多数中文QA场景的均衡点，但你要用你自己的数据验证。"

## 检索层：分块才是RAG的隐藏地基

延迟降了一些，但陈浩知道还不够。检索320ms是当前最大的瓶颈。

他把问题拆开看：320ms的检索，到底慢在哪里？

用`qdrant_client`的API查了一下，搜索本身只有40ms。慢的是什么？慢的是因为他的文档分块策略太粗糙——每个chunk平均2000字，导致向量数量少但每个向量携带信息太多，相似度计算不准确，要检索更多候选才能保证质量，然后Cross-Encoder重排又加了一轮慢计算。

他当时分块的逻辑非常简单：

```python
# 原始分块：按字符数硬切
def naive_chunking(text: str, chunk_size: int = 2000) -> list[str]:
    chunks = []
    for i in range(0, len(text), chunk_size):
        chunks.append(text[i:i + chunk_size])
    return chunks
```

按字符数硬切，这在业界被称为"分块的原罪"。文档里一段完整的逻辑说到一半被切断，下一个chunk从逻辑的中间开始——语义被割裂了。

AI导师给陈浩发了一张分块演进图：

```
Level 1：Fixed-size Chunking（按字符/词数固定切）
         ↓ 语义割裂，信息损失大
Level 2：Semantic Chunking（按句子/段落语义边界切）
         ↓ 边界问题解决了，但没有层次
Level 3：Parent-Child Chunking（大chunk+小chunk双索引）
         ↓ 检索精度高，但没有跨文档摘要能力
Level 4：RAPTOR（递归摘要树，自动生成多级摘要节点）
         ↓ 天然具备跨文档理解能力
```

"你现在在Level 1，要直接跳到Level 4。"AI导师说。

### 父子分块（Parent Document Retriever）

在实现RAPTOR之前，陈浩先理解了父子分块，因为这是RAPTOR的基础逻辑。

父子分块的核心思路是：**小块用于检索，大块用于生成**。

- 小块（child chunk）：200-400字，语义聚焦，向量相似度计算准确
- 大块（parent chunk）：1000-2000字，上下文完整，给LLM推理用

```python
# parent_child_chunker.py
from dataclasses import dataclass
from typing import Optional
import uuid


@dataclass
class Chunk:
    id: str
    content: str
    parent_id: Optional[str]    # None 表示是顶层父chunk
    source: str                  # 来源文件
    metadata: dict


class ParentDocumentChunker:
    """
    父子分块器：
    - 先切大块（parent），保留完整语义上下文
    - 再从大块切小块（child），用于向量检索
    - 检索时找小块，返回时替换为父块内容

    好处：检索精度 + 上下文完整，两者兼顾
    """

    def __init__(
        self,
        parent_chunk_size: int = 1500,  # 父块大小（字符数）
        child_chunk_size: int = 300,    # 子块大小
        overlap: int = 50,              # 子块之间的重叠
    ):
        self.parent_size = parent_chunk_size
        self.child_size = child_chunk_size
        self.overlap = overlap

    def chunk(self, text: str, source: str) -> tuple[list[Chunk], list[Chunk]]:
        """
        返回 (parent_chunks, child_chunks)
        child_chunks 中的 parent_id 指向对应的 parent_chunk
        """
        parents = self._split_to_parents(text, source)
        children = []
        for parent in parents:
            child_chunks = self._split_to_children(parent)
            children.extend(child_chunks)
        return parents, children

    def _split_to_parents(self, text: str, source: str) -> list[Chunk]:
        """按段落边界切父块"""
        paragraphs = text.split('\n\n')
        parents = []
        current = ""
        current_id = str(uuid.uuid4())

        for para in paragraphs:
            if len(current) + len(para) > self.parent_size and current:
                parents.append(Chunk(
                    id=current_id,
                    content=current.strip(),
                    parent_id=None,
                    source=source,
                    metadata={"chunk_type": "parent"},
                ))
                current = para
                current_id = str(uuid.uuid4())
            else:
                current += "\n\n" + para if current else para

        if current:
            parents.append(Chunk(
                id=current_id,
                content=current.strip(),
                parent_id=None,
                source=source,
                metadata={"chunk_type": "parent"},
            ))
        return parents

    def _split_to_children(self, parent: Chunk) -> list[Chunk]:
        """从父块切子块，带重叠"""
        text = parent.content
        children = []
        step = self.child_size - self.overlap

        for i in range(0, len(text), step):
            child_text = text[i:i + self.child_size]
            if len(child_text) < 50:  # 太短的子块跳过
                continue
            children.append(Chunk(
                id=str(uuid.uuid4()),
                content=child_text,
                parent_id=parent.id,
                source=parent.source,
                metadata={"chunk_type": "child"},
            ))
        return children
```

### RAPTOR：递归摘要树

父子分块解决了"检索精度vs上下文"的矛盾，但还是没有解决跨文档理解的问题——两个不同文档的信息，光靠向量检索没办法自动关联。

RAPTOR（Recursive Abstractive Processing for Tree-Organized Retrieval）的核心洞察是：**如果我们主动让LLM对相关文档做摘要，生成的摘要节点天然就包含了跨文档的关联信息**。

陈浩一开始没看懂，AI导师举了个例子："'国庆假期规定'说节假日出差有补贴；'出差报销政策'说特殊日期有额外补贴率。这两个信息在各自文档里都是分散的。RAPTOR把相关文档聚类，让LLM生成一个聚类摘要：'国庆期间出差报销政策'——这个摘要节点直接回答了用户的跨文档问题。"

陈浩开始写RAPTOR实现：

```python
# raptor_indexer.py
import asyncio
from dataclasses import dataclass
from typing import Optional
import numpy as np
from sklearn.cluster import KMeans
from sklearn.preprocessing import normalize


@dataclass
class RaptorNode:
    """RAPTOR树中的一个节点"""
    id: str
    content: str                    # 摘要内容（或原始文档内容）
    level: int                      # 0=原始文档，1=一级摘要，2=二级摘要...
    children_ids: list[str]         # 子节点ID（用于追溯来源）
    embedding: Optional[list[float]] = None


class RaptorIndexer:
    """
    RAPTOR递归摘要树构建器。

    流程：
    1. 输入原始文档chunks（level 0）
    2. 对level 0节点做聚类
    3. 对每个聚类用LLM生成摘要节点（level 1）
    4. 对level 1节点继续聚类，生成level 2摘要
    5. 重复，直到只剩一个根节点

    最终所有层次的节点都放入向量数据库，检索时天然覆盖不同粒度。
    """

    def __init__(self, llm_client, embedder, cluster_size: int = 5):
        self.llm = llm_client
        self.embedder = embedder
        self.cluster_size = cluster_size  # 每个聚类大约多少文档

    async def build_tree(self, documents: list[str]) -> list[RaptorNode]:
        """
        构建RAPTOR树，返回所有节点（原始 + 各层摘要）
        """
        # Level 0：处理原始文档
        level_0_nodes = []
        for i, doc in enumerate(documents):
            embedding = await self.embedder.embed(doc)
            level_0_nodes.append(RaptorNode(
                id=f"L0_{i}",
                content=doc,
                level=0,
                children_ids=[],
                embedding=embedding,
            ))

        all_nodes = list(level_0_nodes)
        current_layer = level_0_nodes

        # 递归构建摘要层
        while len(current_layer) > 1:
            print(f"当前层有 {len(current_layer)} 个节点，开始聚类...")
            next_layer = await self._build_summary_layer(current_layer)
            all_nodes.extend(next_layer)
            current_layer = next_layer
            print(f"生成了 {len(next_layer)} 个摘要节点")

        print(f"RAPTOR树构建完成，共 {len(all_nodes)} 个节点")
        return all_nodes

    async def _build_summary_layer(
        self, nodes: list[RaptorNode]
    ) -> list[RaptorNode]:
        """对一层节点做聚类，为每个聚类生成摘要"""
        if len(nodes) <= 1:
            return nodes

        # 提取向量矩阵做KMeans聚类
        embeddings = np.array([n.embedding for n in nodes])
        embeddings = normalize(embeddings)  # 归一化

        n_clusters = max(2, len(nodes) // self.cluster_size)
        n_clusters = min(n_clusters, len(nodes) - 1)  # 聚类数不能超过节点数

        kmeans = KMeans(n_clusters=n_clusters, random_state=42, n_init=10)
        labels = kmeans.fit_predict(embeddings)

        # 对每个聚类生成摘要
        summary_nodes = []
        for cluster_id in range(n_clusters):
            cluster_nodes = [nodes[i] for i, l in enumerate(labels) if l == cluster_id]
            if not cluster_nodes:
                continue

            summary_content = await self._generate_summary(cluster_nodes)
            summary_embedding = await self.embedder.embed(summary_content)

            level = cluster_nodes[0].level + 1
            summary_node = RaptorNode(
                id=f"L{level}_C{cluster_id}_{len(summary_nodes)}",
                content=summary_content,
                level=level,
                children_ids=[n.id for n in cluster_nodes],
                embedding=summary_embedding,
            )
            summary_nodes.append(summary_node)

        return summary_nodes

    async def _generate_summary(self, nodes: list[RaptorNode]) -> str:
        """用LLM对一个聚类生成综合摘要"""
        combined_text = "\n\n---\n\n".join([n.content for n in nodes])
        prompt = f"""请对以下几段相关内容进行综合摘要，提炼核心信息，保留关键数据和规定。
摘要应该让读者不需要看原文就能理解主要内容。

原文内容：
{combined_text}

综合摘要（300字以内）："""

        response = await self.llm.complete(prompt, temperature=0.1)
        return response.strip()
```

陈浩把RAPTOR构建好后，对公司内部HR文档库重新建了索引。现在向量库里同时存在三层节点：原始文档chunk（level 0）、按主题聚类的摘要（level 1，比如"差旅政策汇总"、"假期规定汇总"）、全局摘要（level 2，"员工福利与管理制度概览"）。

## 查询路由：不同的问题走不同的路

有了语义缓存和RAPTOR索引，陈浩意识到下一个问题：不是所有问题都适合同一套检索策略。

"差旅住宿上限是多少"——这是事实查询，找一个具体的数字，标准向量检索足够。

"出差政策和加班政策哪个更严格"——这是对比查询，需要同时检索两套文档然后比较。

"国庆前后两周出差政策有什么不同"——这是多跳查询，需要先理解"国庆前后"是哪些日期，再去找对应日期的政策。

"介绍一下整个报销流程"——这是摘要型查询，需要把多个文档综合起来给一个全局视图。

如果所有问题都走同一条路，事实查询会被送去做不必要的多跳检索（浪费时间），多跳查询又会被标准检索漏掉关键信息。查询路由（Query Router）就是解决这个问题的。

```python
# query_router.py
from enum import Enum
from dataclasses import dataclass
from typing import Protocol


class QueryType(Enum):
    FACTUAL = "factual"              # 事实查询："住宿上限是多少"
    COMPARISON = "comparison"        # 对比查询："A和B有什么区别"
    MULTI_HOP = "multi_hop"          # 多跳查询："国庆前后政策有什么不同"
    SUMMARIZATION = "summarization"  # 摘要查询："介绍一下报销流程"


@dataclass
class RoutingDecision:
    query_type: QueryType
    strategy_name: str
    reason: str                      # LLM的分类理由（用于debug）
    confidence: float                # 置信度 0-1


class QueryRouter:
    """
    用LLM对查询意图分类，然后路由到不同检索策略。

    分类逻辑：
    - FACTUAL：包含"多少/几/是什么/具体"等词，或问具体数值/规定
    - COMPARISON：包含"区别/不同/哪个更/比较"等词
    - MULTI_HOP：包含时间范围/因果关联，需要先推理再检索
    - SUMMARIZATION：包含"介绍/说明/概述/整体"等词
    """

    CLASSIFICATION_PROMPT = """请分析以下用户问题的查询类型，从四类中选择最合适的一个：

FACTUAL：用户在询问某个具体的事实/数值/规定
COMPARISON：用户在比较两个或多个事物的区别
MULTI_HOP：用户的问题需要多步推理，例如先确定时间/条件，再找对应规定
SUMMARIZATION：用户想了解某个主题的整体概况

用户问题：{query}

请用JSON格式回答：
{{
  "type": "FACTUAL|COMPARISON|MULTI_HOP|SUMMARIZATION",
  "reason": "一句话解释分类理由",
  "confidence": 0.0-1.0
}}"""

    def __init__(self, llm_client):
        self.llm = llm_client

    async def classify(self, query: str) -> RoutingDecision:
        """用LLM分类查询类型"""
        import json
        prompt = self.CLASSIFICATION_PROMPT.format(query=query)

        try:
            response = await self.llm.complete(prompt, temperature=0.0)
            parsed = json.loads(response)
            query_type = QueryType(parsed["type"].lower())
            return RoutingDecision(
                query_type=query_type,
                strategy_name=query_type.value,
                reason=parsed.get("reason", ""),
                confidence=float(parsed.get("confidence", 0.8)),
            )
        except Exception as e:
            # 降级：无法分类时默认走FACTUAL
            return RoutingDecision(
                query_type=QueryType.FACTUAL,
                strategy_name="factual",
                reason=f"分类失败，降级处理: {e}",
                confidence=0.5,
            )

    async def route(self, query: str) -> "RetrievalStrategy":
        """根据分类结果返回对应的检索策略"""
        decision = await self.classify(query)

        strategy_map = {
            QueryType.FACTUAL: StandardRetrievalStrategy(top_k=5),
            QueryType.COMPARISON: ComparisonRetrievalStrategy(top_k_per_entity=5),
            QueryType.MULTI_HOP: MultiHopRetrievalStrategy(max_hops=3),
            QueryType.SUMMARIZATION: SummarizationRetrievalStrategy(top_k=15),
        }

        strategy = strategy_map[decision.query_type]
        strategy.routing_decision = decision  # 附上路由决策，用于可观测性
        return strategy


class StandardRetrievalStrategy:
    """标准向量检索：用于事实查询"""
    def __init__(self, top_k: int = 5):
        self.top_k = top_k
        self.routing_decision: RoutingDecision = None


class ComparisonRetrievalStrategy:
    """对比检索：分别检索每个实体，然后合并"""
    def __init__(self, top_k_per_entity: int = 5):
        self.top_k_per_entity = top_k_per_entity
        self.routing_decision: RoutingDecision = None


class MultiHopRetrievalStrategy:
    """多跳检索：支持迭代式检索"""
    def __init__(self, max_hops: int = 3):
        self.max_hops = max_hops
        self.routing_decision: RoutingDecision = None


class SummarizationRetrievalStrategy:
    """摘要检索：检索更多文档，优先RAPTOR高层节点"""
    def __init__(self, top_k: int = 15):
        self.top_k = top_k
        self.routing_decision: RoutingDecision = None
```

## 多跳推理：当答案需要跨越两篇文档

查询路由实现完，MultiHopRetrievalStrategy还只是个空壳。陈浩着手实现核心逻辑。

多跳推理（Multi-hop Reasoning）的核心是：**不是一次检索就完，而是用第一次检索的结果来指导第二次检索**。

就像人类回答"国庆前后出差政策有什么不同"的思考过程：
1. 先想"国庆是几号到几号"（检索第一跳：假期安排）
2. 确定日期后，再去查"这些日期前后的出差报销规定"（检索第二跳：差旅政策）
3. 把两次结果综合，给出完整答案

```python
# multi_hop_retriever.py
from dataclasses import dataclass, field


@dataclass
class HopResult:
    """单次检索的结果"""
    hop_number: int
    query_used: str         # 本次检索用的查询
    documents_found: list   # 找到的文档
    extracted_facts: str    # 从文档中提取的关键信息


@dataclass
class MultiHopResult:
    """多跳检索的完整结果"""
    original_query: str
    hops: list[HopResult] = field(default_factory=list)
    final_context: str = ""


class MultiHopRetriever:
    """
    迭代式多跳检索器。

    算法：
    1. 用原始问题检索第一批文档
    2. 让LLM分析：这些文档够了吗？如果不够，需要再搜什么？
    3. 若LLM说需要继续搜，用新查询做第二跳
    4. 最多跳max_hops次，防止无限循环

    关键：每跳的检索结果都会告知下一跳应该查什么
    """

    def __init__(self, retriever, llm_client, embedder, max_hops: int = 3):
        self.retriever = retriever
        self.llm = llm_client
        self.embedder = embedder
        self.max_hops = max_hops

    async def retrieve(self, query: str) -> MultiHopResult:
        result = MultiHopResult(original_query=query)
        accumulated_facts = ""

        for hop in range(self.max_hops):
            # 第一跳用原始问题，后续跳用LLM生成的跟进查询
            if hop == 0:
                search_query = query
            else:
                search_query = await self._generate_followup_query(
                    original_query=query,
                    accumulated_facts=accumulated_facts,
                )
                if search_query is None:
                    # LLM认为已经收集到足够信息
                    break

            # 执行检索
            query_embedding = await self.embedder.embed(search_query)
            docs = await self.retriever.search(query_embedding, top_k=5)

            # 从文档中提取关键事实
            extracted = await self._extract_facts(query, docs)
            accumulated_facts += f"\n\n[第{hop+1}次检索结果]\n{extracted}"

            hop_result = HopResult(
                hop_number=hop + 1,
                query_used=search_query,
                documents_found=docs,
                extracted_facts=extracted,
            )
            result.hops.append(hop_result)

        result.final_context = accumulated_facts
        return result

    async def _generate_followup_query(
        self, original_query: str, accumulated_facts: str
    ) -> str | None:
        """让LLM判断是否需要更多信息，如果需要生成跟进查询"""
        prompt = f"""原始问题：{original_query}

已收集的信息：
{accumulated_facts}

请判断：
1. 已收集的信息是否足以完整回答原始问题？
2. 如果不足，还需要查询什么？

请用JSON回答：
{{
  "sufficient": true/false,
  "followup_query": "如果not sufficient，填写需要继续查询的具体问题；如果sufficient填null"
}}"""

        import json
        try:
            response = await self.llm.complete(prompt, temperature=0.0)
            parsed = json.loads(response)
            if parsed.get("sufficient", True):
                return None  # 信息已足够，停止跳转
            return parsed.get("followup_query")
        except Exception:
            return None  # 解析失败，停止跳转

    async def _extract_facts(self, query: str, docs: list) -> str:
        """从检索结果中提取与问题相关的关键事实"""
        if not docs:
            return "（未找到相关文档）"

        combined = "\n\n".join([d.page_content for d in docs[:3]])  # 最多取3个
        prompt = f"""从以下文档中，提取与问题【{query}】直接相关的关键事实和数据。
只提取事实，不要推理，不要总结。

文档：
{combined}

关键事实（如无相关内容，说明"未找到相关内容"）："""

        return await self.llm.complete(prompt, temperature=0.0)
```

陈浩测试了"国庆前后两周出差政策有什么不同"：

## 上下文压缩：少喂给LLM

RAG检索回来10个文档，不是每一段都对当前问题有价值。把10个文档全部塞给LLM，既浪费token，又让LLM在大量无关信息中迷失。

上下文压缩（Context Compression）的逻辑是：**在传给LLM之前，先让另一个（便宜的）LLM过滤掉与问题无关的部分**。

```python
# context_compressor.py
from dataclasses import dataclass


@dataclass
class CompressedDocument:
    original_source: str
    original_length: int
    compressed_content: str
    compressed_length: int
    compression_ratio: float


class ContextualCompressor:
    """
    上下文压缩器：提取每个检索文档中与问题相关的部分。

    原理：
    - 检索到的文档可能很长，大部分内容与当前问题无关
    - 用一个快速/便宜的LLM（gpt-4o-mini）先做过滤
    - 只把相关部分传给主LLM（gpt-4o）

    效果：
    - 主LLM的上下文长度减少40%-60%
    - 噪音减少，答案质量反而提升
    - 主LLM调用成本降低
    """

    COMPRESS_PROMPT = """从以下文档中，提取与问题【{query}】直接相关的内容片段。

要求：
- 只保留与问题直接相关的句子/段落
- 保持原文措辞，不要改写
- 如果文档与问题完全无关，回复"IRRELEVANT"
- 不要添加任何解释或总结

文档内容：
{document}

相关内容（或"IRRELEVANT"）："""

    def __init__(self, llm_client):
        self.llm = llm_client  # 建议用便宜的小模型

    async def compress(
        self,
        query: str,
        documents: list,
    ) -> list[CompressedDocument]:
        """压缩所有文档，过滤无关内容"""
        tasks = [self._compress_single(query, doc) for doc in documents]
        results = await asyncio.gather(*tasks, return_exceptions=True)

        compressed = []
        for result in results:
            if isinstance(result, Exception):
                continue  # 压缩失败的跳过
            if result is not None:
                compressed.append(result)
        return compressed

    async def _compress_single(
        self, query: str, doc
    ) -> CompressedDocument | None:
        """压缩单个文档"""
        original_content = doc.page_content
        prompt = self.COMPRESS_PROMPT.format(
            query=query,
            document=original_content[:3000]  # 限制单文档长度
        )

        compressed_content = await self.llm.complete(prompt, temperature=0.0)

        if "IRRELEVANT" in compressed_content.upper():
            return None  # 这个文档与问题无关，丢弃

        return CompressedDocument(
            original_source=doc.metadata.get("source", "unknown"),
            original_length=len(original_content),
            compressed_content=compressed_content.strip(),
            compressed_length=len(compressed_content.strip()),
            compression_ratio=len(compressed_content) / len(original_content),
        )
```

陈浩在测试中发现，经过上下文压缩后，传给主LLM的上下文平均长度从4200字缩短到了1600字，节省了62%。推理质量反而略有提升——因为主LLM不再被大量无关信息干扰。

## Embedding缓存：减少重复计算

语义缓存解决了"同一个问题问多次"的问题，但还有一个隐藏的重复计算场景：**增量索引时对新文档做Embedding，以及对热门词汇/短语的Embedding**。

公司内部文档经常提到"差旅费"、"发票"、"报销"这些词，每次分块时这些词都会出现在不同的chunk里，Embedding时反复计算相同的短语。

更直接的优化是对查询做Embedding缓存——虽然每次查询都不同，但高频查询（"差旅怎么报销"、"住宿上限"）会被反复提问。

```python
# embedding_cache.py
import hashlib
import json
from typing import Optional
import asyncio


class EmbeddingCache:
    """
    Embedding结果缓存，减少重复的API调用。

    策略：
    - 内存LRU缓存（热数据，毫秒级命中）
    - Redis持久化（跨重启、跨实例共享，10ms级命中）
    - 预热：系统启动时对高频查询批量预计算
    """

    def __init__(self, redis_client=None, max_memory_entries: int = 5000):
        self._memory: dict[str, list[float]] = {}
        self._access_order: list[str] = []
        self._max_memory = max_memory_entries
        self.redis = redis_client
        self._lock = asyncio.Lock()

        # 统计
        self._memory_hits = 0
        self._redis_hits = 0
        self._misses = 0

    def _cache_key(self, text: str) -> str:
        """用MD5作为缓存key（Embedding的输入是精确文本，用精确hash合适）"""
        return f"emb:{hashlib.md5(text.encode()).hexdigest()}"

    async def get(self, text: str) -> Optional[list[float]]:
        key = self._cache_key(text)

        # 先查内存
        async with self._lock:
            if key in self._memory:
                self._memory_hits += 1
                # LRU：移到末尾
                self._access_order.remove(key)
                self._access_order.append(key)
                return self._memory[key]

        # 再查Redis
        if self.redis:
            cached = await self.redis.get(key)
            if cached:
                self._redis_hits += 1
                embedding = json.loads(cached)
                await self._store_in_memory(key, embedding)
                return embedding

        self._misses += 1
        return None

    async def set(self, text: str, embedding: list[float]) -> None:
        key = self._cache_key(text)
        await self._store_in_memory(key, embedding)
        if self.redis:
            await self.redis.setex(
                key,
                86400 * 7,  # 7天过期
                json.dumps(embedding),
            )

    async def _store_in_memory(self, key: str, embedding: list[float]) -> None:
        async with self._lock:
            if len(self._memory) >= self._max_memory:
                # LRU淘汰：移除最久未访问的
                oldest = self._access_order.pop(0)
                self._memory.pop(oldest, None)
            self._memory[key] = embedding
            self._access_order.append(key)

    async def warmup(self, frequent_queries: list[str], embedder) -> None:
        """预热：批量计算高频查询的Embedding"""
        print(f"开始预热 {len(frequent_queries)} 个高频查询...")
        miss_queries = []
        for query in frequent_queries:
            if await self.get(query) is None:
                miss_queries.append(query)

        if miss_queries:
            # 批量Embedding（减少API调用次数）
            batch_size = 20
            for i in range(0, len(miss_queries), batch_size):
                batch = miss_queries[i:i + batch_size]
                embeddings = await embedder.embed_batch(batch)
                for query, emb in zip(batch, embeddings):
                    await self.set(query, emb)

        print(f"预热完成，{len(miss_queries)} 个查询已缓存")

    def stats(self) -> dict:
        total = self._memory_hits + self._redis_hits + self._misses
        return {
            "memory_hits": self._memory_hits,
            "redis_hits": self._redis_hits,
            "misses": self._misses,
            "total_requests": total,
            "memory_hit_rate": f"{self._memory_hits / max(total, 1):.1%}",
            "overall_hit_rate": f"{(self._memory_hits + self._redis_hits) / max(total, 1):.1%}",
        }
```

## 增量索引：文档更新时的高效处理

HR部门每个月都会更新差旅报销标准，法务部偶尔会修改合规文档。每次文档更新，如果重新对整个知识库做Embedding，代价太大。

增量索引（Incremental Indexing）的核心思路是：**用文件内容的哈希值判断文档是否变化，只重新索引有变化的文档**。

```python
# incremental_indexer.py
import hashlib
import asyncio
from dataclasses import dataclass
from pathlib import Path


@dataclass
class IndexUpdateResult:
    file_path: str
    status: str          # "unchanged" | "updated" | "deleted" | "new"
    chunks_updated: int
    time_ms: float


class IncrementalIndexer:
    """
    高效增量索引器：只处理变化的文档。

    核心机制：
    - 用SHA256 hash检测文档是否变化
    - 变化时：删除旧向量 + 重新Embedding + 存新向量
    - 未变化：跳过，零计算成本

    存储：
    - hash记录存 PostgreSQL（file_path → content_hash + indexed_at）
    - 向量存 Qdrant（按 source metadata 筛选删除）
    """

    def __init__(self, db_client, vector_store, processor, chunker, embedder):
        self.db = db_client
        self.vector_store = vector_store
        self.processor = processor
        self.chunker = chunker
        self.embedder = embedder

    async def upsert(self, file_path: str) -> IndexUpdateResult:
        """
        对单个文件做增量更新。

        流程：
        1. 计算文件SHA256
        2. 查DB中存的旧hash
        3. 如果相同：返回 unchanged，零操作
        4. 如果不同：删旧索引，建新索引，更新hash
        """
        import time
        start = time.time()

        new_hash = await self._compute_hash(file_path)
        existing_hash = await self.db.get_document_hash(file_path)

        if existing_hash == new_hash:
            return IndexUpdateResult(
                file_path=file_path,
                status="unchanged",
                chunks_updated=0,
                time_ms=(time.time() - start) * 1000,
            )

        is_new = existing_hash is None

        # 删除旧索引（如果存在）
        if not is_new:
            deleted = await self.vector_store.delete_by_metadata(
                filter={"source": file_path}
            )
            print(f"  删除旧向量: {deleted} 条")

        # 处理新文档
        raw_docs = await self.processor.load(file_path)
        chunks = await self.chunker.chunk(raw_docs)
        embeddings = await self._embed_batch(chunks)

        # 写入向量库
        await self.vector_store.add(embeddings, metadata_source=file_path)

        # 更新hash记录
        await self.db.save_document_hash(file_path, new_hash)

        return IndexUpdateResult(
            file_path=file_path,
            status="new" if is_new else "updated",
            chunks_updated=len(chunks),
            time_ms=(time.time() - start) * 1000,
        )

    async def sync_directory(self, dir_path: str) -> list[IndexUpdateResult]:
        """
        同步整个目录：
        - 新增/修改的文件：upsert
        - 已删除的文件：从向量库删除
        """
        current_files = set(
            str(p) for p in Path(dir_path).rglob("*.txt")
            if p.is_file()
        )
        indexed_files = set(await self.db.list_indexed_files(dir_path))

        # 删除已不存在的文件
        deleted_files = indexed_files - current_files
        delete_tasks = [self._remove_file(f) for f in deleted_files]

        # 更新/新增文件
        upsert_tasks = [self.upsert(f) for f in current_files]

        results = await asyncio.gather(*upsert_tasks, return_exceptions=True)
        await asyncio.gather(*delete_tasks, return_exceptions=True)

        return [r for r in results if isinstance(r, IndexUpdateResult)]

    async def _compute_hash(self, file_path: str) -> str:
        content = Path(file_path).read_bytes()
        return hashlib.sha256(content).hexdigest()

    async def _embed_batch(self, chunks: list, batch_size: int = 50) -> list:
        """批量Embedding，每批50个，避免API限流"""
        all_embeddings = []
        for i in range(0, len(chunks), batch_size):
            batch = chunks[i:i + batch_size]
            texts = [c.content for c in batch]
            embeddings = await self.embedder.embed_batch(texts)
            for chunk, emb in zip(batch, embeddings):
                all_embeddings.append({
                    "id": chunk.id,
                    "vector": emb,
                    "payload": {
                        "content": chunk.content,
                        "source": chunk.source,
                        "metadata": chunk.metadata,
                    }
                })
        return all_embeddings

    async def _remove_file(self, file_path: str) -> None:
        await self.vector_store.delete_by_metadata({"source": file_path})
        await self.db.delete_document_hash(file_path)
```

## RAG可观测性：让每一步都留下痕迹

```python
# rag_observer.py
from dataclasses import dataclass, field
from typing import Optional
import time
import json


@dataclass
class RAGTrace:
    """一次完整RAG请求的全链路追踪"""
    # 请求基础信息
    trace_id: str
    query: str
    timestamp: str

    # 语义缓存
    cache_hit: bool = False
    cache_similarity: float = 0.0        # 命中时的相似度

    # Embedding
    embed_latency_ms: float = 0.0
    embed_cache_hit: bool = False

    # 查询路由
    query_type: str = ""                 # FACTUAL/COMPARISON/MULTI_HOP/SUMMARIZATION
    routing_confidence: float = 0.0

    # 检索
    retrieval_latency_ms: float = 0.0
    retrieved_doc_count: int = 0
    raptor_levels_hit: list[int] = field(default_factory=list)  # 命中的RAPTOR层级
    semantic_scores: list[float] = field(default_factory=list)   # 检索相似度列表

    # 多跳推理
    hop_count: int = 0                   # 实际跳数（FACTUAL时为1）

    # 上下文压缩
    context_before_compression: int = 0  # 压缩前总字数
    context_after_compression: int = 0   # 压缩后总字数

    # LLM推理
    rerank_latency_ms: float = 0.0
    llm_latency_ms: float = 0.0
    prompt_tokens: int = 0
    completion_tokens: int = 0

    # 质量指标
    faithfulness_score: Optional[float] = None  # 采样评估（不是每次都有）

    # 总耗时
    total_latency_ms: float = 0.0


class RAGObserver:
    """
    RAG全链路可观测性收集器。

    支持两种后端：
    - Prometheus + Grafana：指标可视化
    - 日志（结构化JSON）：问题排查

    关键指标：
    - rag.cache.hit_rate（语义缓存命中率）
    - rag.latency.p99（总延迟P99）
    - rag.retrieval.avg_scores（检索相似度均值）
    - rag.query_type.distribution（查询类型分布）
    - rag.hop.avg_count（多跳平均跳数）
    """

    def __init__(self, metrics_client=None, logger=None):
        self.metrics = metrics_client
        self.logger = logger
        self._traces: list[RAGTrace] = []

    def record(self, trace: RAGTrace) -> None:
        """记录一次请求的追踪数据"""
        self._traces.append(trace)

        if self.logger:
            self.logger.info("rag_trace", extra={"trace": self._to_dict(trace)})

        if self.metrics:
            self._emit_metrics(trace)

    def _emit_metrics(self, trace: RAGTrace) -> None:
        """向Prometheus/StatsD发送指标"""
        m = self.metrics

        # 延迟分布（Histogram）
        m.histogram("rag.latency.total", trace.total_latency_ms)
        m.histogram("rag.latency.embed", trace.embed_latency_ms)
        m.histogram("rag.latency.retrieval", trace.retrieval_latency_ms)
        m.histogram("rag.latency.llm", trace.llm_latency_ms)

        # 缓存（Counter）
        if trace.cache_hit:
            m.counter("rag.cache.hit")
        else:
            m.counter("rag.cache.miss")

        if trace.embed_cache_hit:
            m.counter("rag.embed_cache.hit")
        else:
            m.counter("rag.embed_cache.miss")

        # 查询类型分布
        m.counter(f"rag.query_type.{trace.query_type.lower()}")

        # 检索质量
        if trace.semantic_scores:
            avg_score = sum(trace.semantic_scores) / len(trace.semantic_scores)
            m.gauge("rag.retrieval.avg_similarity", avg_score)
            m.gauge("rag.retrieval.top_similarity", max(trace.semantic_scores))

        # 多跳
        m.histogram("rag.hop.count", trace.hop_count)

        # 上下文压缩率
        if trace.context_before_compression > 0:
            ratio = trace.context_after_compression / trace.context_before_compression
            m.gauge("rag.context.compression_ratio", ratio)

        # Token消耗
        m.counter("rag.tokens.prompt", trace.prompt_tokens)
        m.counter("rag.tokens.completion", trace.completion_tokens)

        # Faithfulness（采样评估，不是每次都有）
        if trace.faithfulness_score is not None:
            m.gauge("rag.quality.faithfulness", trace.faithfulness_score)

    def summary(self, last_n: int = 100) -> dict:
        """返回最近N次请求的聚合统计"""
        recent = self._traces[-last_n:]
        if not recent:
            return {}

        latencies = [t.total_latency_ms for t in recent]
        latencies.sort()

        cache_hits = sum(1 for t in recent if t.cache_hit)
        query_types = {}
        for t in recent:
            qt = t.query_type or "UNKNOWN"
            query_types[qt] = query_types.get(qt, 0) + 1

        return {
            "requests": len(recent),
            "cache_hit_rate": f"{cache_hits / len(recent):.1%}",
            "latency_p50": f"{latencies[len(latencies)//2]:.0f}ms",
            "latency_p99": f"{latencies[int(len(latencies)*0.99)]:.0f}ms",
            "query_type_distribution": query_types,
            "avg_hop_count": sum(t.hop_count for t in recent) / len(recent),
            "avg_docs_retrieved": sum(t.retrieved_doc_count for t in recent) / len(recent),
        }

    def _to_dict(self, trace: RAGTrace) -> dict:
        return {
            "trace_id": trace.trace_id,
            "query": trace.query[:100],  # 截断，避免日志过大
            "cache_hit": trace.cache_hit,
            "query_type": trace.query_type,
            "total_latency_ms": trace.total_latency_ms,
            "hop_count": trace.hop_count,
            "faithfulness": trace.faithfulness_score,
        }
```
