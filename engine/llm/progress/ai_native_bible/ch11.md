## Tool系统设计原则

那天晚上，陈浩回到家，没有急着写代码，而是在笔记本上把当天白板的内容展开写了一遍。

他问自己：一个"工具"到底是什么？

在他原来的实现里，工具就是一个Python函数，加上一个字符串描述，打包成JSON告诉GPT。但现在他意识到，一个工具应该包含更多的信息：

- **名字**：唯一标识符
- **描述**：告诉AI什么时候用这个工具（这个描述至关重要）
- **参数**：JSON Schema，告诉AI怎么用
- **实现**：具体的函数逻辑
- **安全级别**：READ / WRITE / DANGEROUS
- **缓存策略**：只读工具的结果能不能缓存，缓存多久
- **是否需要确认**：DANGEROUS级别的工具，执行前要不要让人审批

其中有两件事，陈浩之前完全没想过：**描述的精确性**和**缓存**。

描述的精确性，是AI行为的核心决定因素之一。

陈浩打开他的query_expenses工具，看了一眼描述：`"查询报销记录"`。

就这六个字。

他想起架构导师说过另一句话（陈浩觉得这话更接近AI导师的风格，但架构导师偶尔也会说）："你是在给AI提供文档，不是在给人看。"

一个人看到"查询报销记录"，可以根据上下文猜出这个工具是干什么的。但AI做的是统计匹配——它根据用户意图和工具描述的语义相似度来决定调用哪个工具。描述越模糊，AI越需要猜，猜错的概率越高。

缓存，是工程效率问题。

如果用户问了三次"我上个月的报销总额是多少"，你的Agent每次都去数据库查询一次，不仅慢，还浪费。只读工具的结果是可以缓存的——同样的参数，返回同样的结果，这是幂等性（idempotent）的定义，Java工程师陈浩对这个词不陌生。

## 可扩展的Tool注册系统

写完笔记，陈浩打开编辑器，开始重构他的工具系统。

第一步：定义数据结构。

```python
# tools/registry.py

from dataclasses import dataclass, field
from enum import Enum
from typing import Callable, Any
import inspect
import json
import time
import asyncio


class ToolSafety(Enum):
    READ = "read"         # 只读，安全，可缓存
    WRITE = "write"       # 写操作，有副作用
    DANGEROUS = "dangerous"  # 高风险，执行前需二次确认


@dataclass
class ToolSpec:
    """工具的完整规格定义"""
    name: str
    description: str
    parameters: dict          # JSON Schema格式
    func: Callable
    safety: ToolSafety = ToolSafety.READ
    requires_confirmation: bool = False
    cache_ttl: int = 0        # 缓存时间（秒），0表示不缓存
    tags: list[str] = field(default_factory=list)
```

这是工具的"身份证"。每个工具不再只是一个函数，而是一个携带完整元数据的数据类。

第二步：实现注册表。

```python
class ToolRegistry:
    """
    工具注册表：负责工具的注册、发现和执行。
    类比 Spring 的 ApplicationContext，但轻量得多。
    """

    def __init__(self):
        self._tools: dict[str, ToolSpec] = {}
        self._cache: dict[str, tuple[Any, float]] = {}  # (result, expire_at)

    def register(self, spec: ToolSpec) -> None:
        """手动注册一个工具规格"""
        if spec.name in self._tools:
            raise ValueError(f"工具 '{spec.name}' 已存在，禁止重复注册")
        # DANGEROUS 级别的工具自动设置 requires_confirmation
        if spec.safety == ToolSafety.DANGEROUS:
            spec.requires_confirmation = True
        self._tools[spec.name] = spec

    def tool(
        self,
        name: str,
        description: str,
        parameters: dict | None = None,
        safety: ToolSafety = ToolSafety.READ,
        cache_ttl: int = 0,
        tags: list[str] | None = None,
    ):
        """
        装饰器方式注册工具。
        
        用法：
            @registry.tool(
                name="query_expenses",
                description="查询指定用户在指定时间范围内的差旅报销记录",
                safety=ToolSafety.READ,
                cache_ttl=300,
            )
            async def query_expenses(user_id: str, start_date: str) -> list:
                ...
        """
        def decorator(func: Callable) -> Callable:
            # 如果没有手动提供 parameters，从函数签名自动推断
            schema = parameters or self._infer_schema(func)
            spec = ToolSpec(
                name=name,
                description=description,
                parameters=schema,
                func=func,
                safety=safety,
                cache_ttl=cache_ttl,
                tags=tags or [],
            )
            self.register(spec)
            return func  # 返回原函数，不破坏调用方式
        return decorator

    def _infer_schema(self, func: Callable) -> dict:
        """从函数类型注解推断 JSON Schema（简化版）"""
        sig = inspect.signature(func)
        properties = {}
        required = []
        type_map = {
            str: "string",
            int: "integer",
            float: "number",
            bool: "boolean",
            list: "array",
            dict: "object",
        }
        for param_name, param in sig.parameters.items():
            annotation = param.annotation
            if annotation is inspect.Parameter.empty:
                json_type = "string"
            else:
                # 处理 Optional[str] -> str 的情况
                origin = getattr(annotation, "__origin__", None)
                if origin is type(None):
                    json_type = "string"
                else:
                    json_type = type_map.get(annotation, "string")
            
            properties[param_name] = {"type": json_type}
            
            # 没有默认值的参数是 required
            if param.default is inspect.Parameter.empty:
                required.append(param_name)

        return {
            "type": "object",
            "properties": properties,
            "required": required,
        }

    def get_openai_tools(self, safety_filter: list[ToolSafety] | None = None) -> list[dict]:
        """
        转换为 OpenAI tools 格式，发送给 LLM。
        safety_filter: 只返回指定安全级别的工具（None=全部）
        """
        result = []
        for spec in self._tools.values():
            if safety_filter and spec.safety not in safety_filter:
                continue
            result.append({
                "type": "function",
                "function": {
                    "name": spec.name,
                    "description": spec.description,
                    "parameters": spec.parameters,
                },
            })
        return result

    async def execute(self, tool_name: str, args: dict) -> Any:
        """
        执行工具，处理缓存逻辑。
        """
        if tool_name not in self._tools:
            raise KeyError(f"工具 '{tool_name}' 未注册")
        
        spec = self._tools[tool_name]

        # DANGEROUS 工具检查
        if spec.requires_confirmation:
            raise PermissionError(
                f"工具 '{tool_name}' 是 DANGEROUS 级别，"
                f"需要通过 execute_with_confirmation() 调用"
            )

        # 缓存检查（只对 READ 工具生效）
        if spec.safety == ToolSafety.READ and spec.cache_ttl > 0:
            cache_key = f"{tool_name}:{json.dumps(args, sort_keys=True)}"
            if cache_key in self._cache:
                cached_result, expire_at = self._cache[cache_key]
                if time.time() < expire_at:
                    return cached_result  # 命中缓存，直接返回

        # 调用真正的工具函数
        if asyncio.iscoroutinefunction(spec.func):
            result = await spec.func(**args)
        else:
            result = spec.func(**args)

        # 写入缓存
        if spec.safety == ToolSafety.READ and spec.cache_ttl > 0:
            cache_key = f"{tool_name}:{json.dumps(args, sort_keys=True)}"
            self._cache[cache_key] = (result, time.time() + spec.cache_ttl)

        return result

    async def execute_with_confirmation(
        self,
        tool_name: str,
        args: dict,
        confirmed: bool = False,
    ) -> Any:
        """专门用于执行 DANGEROUS 级别工具，需要显式确认"""
        if not confirmed:
            spec = self._tools.get(tool_name)
            raise PermissionError(
                f"工具 '{tool_name}' 需要确认。"
                f"操作内容：{spec.description}，参数：{args}。"
                f"确认后请传入 confirmed=True。"
            )
        spec = self._tools[tool_name]
        if asyncio.iscoroutinefunction(spec.func):
            return await spec.func(**args)
        return spec.func(**args)

    def list_tools(self, safety: ToolSafety | None = None) -> list[ToolSpec]:
        """列出已注册的工具，支持按安全级别过滤"""
        if safety is None:
            return list(self._tools.values())
        return [s for s in self._tools.values() if s.safety == safety]
```

第三步：使用装饰器注册工具。这是最优雅的使用方式：

```python

    ),

)

    user_id: str,
    order_id: str,
    amount: float,
```

陈浩对比新旧两种写法，感受到了明显的差异。

旧的方式：函数定义在一处，注册代码在另一处，描述是事后补上的六个字，安全分类完全不存在。

新的方式：装饰器把"这是什么工具、怎么用、有多危险"全部附着在函数定义旁边，一眼能看清。

这让他想起在Java里用Spring的`@Service`、`@Repository`的感觉——类型即文档，注解即配置。

## 动态Tool加载：插件化

工具注册表做好之后，陈浩遇到了第二个问题：当工具越来越多，他不希望在一个文件里导入所有工具模块。

他想要的是：把工具文件放到一个目录里，系统自动发现并加载，就像插件一样。

```python
# tools/loader.py

import importlib
import importlib.util
from pathlib import Path
from tools.registry import ToolRegistry


def load_tools_from_directory(registry: ToolRegistry, tools_dir: str) -> int:
    """
    从指定目录自动发现并加载所有工具模块。
    规则：目录下所有 *_tools.py 文件都会被导入。
    
    目录结构示例：
        tools/
            expense_tools.py    ← 报销相关工具
            crm_tools.py        ← CRM相关工具
            notification_tools.py ← 通知工具
            loader.py           ← 本文件，不加载
    
    返回：加载的工具数量
    """
    tools_path = Path(tools_dir)
    if not tools_path.is_dir():
        raise FileNotFoundError(f"工具目录不存在: {tools_dir}")

    loaded_count = 0
    
    for file in sorted(tools_path.glob("*_tools.py")):
        module_name = file.stem  # 文件名（不含扩展名）
        
        spec = importlib.util.spec_from_file_location(module_name, file)
        if spec is None or spec.loader is None:
            print(f"[WARN] 无法加载模块: {file}")
            continue
        
        module = importlib.util.module_from_spec(spec)
        
        # 注入 registry，让工具模块能使用同一个注册表
        module.registry = registry
        
        try:
            spec.loader.exec_module(module)
            # 统计本模块新注册的工具数量
            count = len([
                name for name, tool in registry._tools.items()
                if not hasattr(module, "__loaded__")
            ])
            print(f"[OK] 加载 {module_name}: {file.name}")
            loaded_count += 1
        except Exception as e:
            print(f"[ERROR] 加载 {file.name} 失败: {e}")
    
    return loaded_count


# 使用示例
if __name__ == "__main__":
    from tools.registry import ToolRegistry

    registry = ToolRegistry()
    n = load_tools_from_directory(registry, "./tools")
    
    print(f"\n已注册工具总数：{len(registry._tools)}")
    for spec in registry.list_tools():
        safety_icon = {"read": "📖", "write": "✏️", "dangerous": "⚠️"}
        print(f"  {safety_icon.get(spec.safety.value, '?')} {spec.name}")
```

插件化之后，新增一个工具只需要：

1. 在`tools/`目录下创建一个新的`xxx_tools.py`文件
2. 用`@registry.tool(...)`装饰器注册工具函数
3. 重启服务，工具自动被发现和加载

核心逻辑不需要修改，完全符合开闭原则（Open/Closed Principle）——对扩展开放，对修改关闭。

## MCP协议：工具的USB接口标准

那周周末，陈浩在Anthropic的文档站上找资料，偶然发现了一个词：**MCP**，全称 Model Context Protocol。

他花了十分钟读完了介绍文档，然后坐在椅背上发了两分钟呆。

不是因为难——而是因为太准了。

他一直觉得自己工具系统有个更深层的问题：这些工具被硬绑在一个特定的Agent里，被一个特定的Python进程调用，被一个特定的OpenAI客户端使用。如果他想在另一个项目里用同样的工具，他只能复制代码。如果他想让Claude Desktop也能用这些工具，他没有办法。

MCP解决的就是这个问题。

MCP是什么？一句话：**工具的USB接口标准**。

USB接口之前，每种设备有自己的接口——鼠标插PS/2、存储插并口、调制解调器插COM口。USB出现后，所有设备都用一种接口，设备变成了可插拔的标准件。

MCP对AI工具做的是同一件事。

在MCP之前，每个AI应用都自己定义工具格式。OpenAI有一套函数调用格式，Anthropic有一套tool_use格式，各种框架又各有各的包装。工具写了只能用在特定的AI应用里。

MCP定义了一个标准协议：工具提供方（MCP Server）和工具使用方（MCP Client）之间用JSON-RPC通信，协议格式固定，与具体的AI模型无关。

这意味着：你按MCP协议实现的工具服务，可以被任何支持MCP的AI应用使用——Claude Desktop、Cursor、自己写的Agent，都可以。

MCP协议的设计哲学体现在三个词：

**标准化**（Standard）：消息格式、工具描述格式、错误格式都有规范，不再是"每个框架各搞一套"。

**隔离性**（Isolation）：工具服务（MCP Server）和AI应用（MCP Client）是独立进程，工具的代码更新不需要重启AI应用，AI应用崩溃也不影响工具服务。

**可组合性**（Composability）：一个AI应用可以连接多个MCP Server，每个Server专注一个领域。报销系统、CRM系统、通知系统各是一个MCP Server，Agent按需连接。

## MCP Client集成

MCP Server解决了"让工具变成独立服务"的问题。但陈浩还需要在他的自定义Agent里能调用MCP Server——换句话说，他需要实现一个MCP Client。

```python
# agent/mcp_client.py
"""
MCP Client：在 Agent 中连接并使用 MCP Server 提供的工具
"""

import json
import asyncio
from contextlib import asynccontextmanager
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client


class MCPToolClient:
    """
    封装 MCP Client，让 Agent 可以像使用本地工具一样使用 MCP Server 的工具。
    
    使用示例：
        client = MCPToolClient()
        await client.connect("python mcp_servers/expense_server.py")
        
        tools = await client.list_tools()  # 获取 OpenAI tools 格式
        result = await client.call_tool("query_expenses", {"user_id": "EMP-001"})
        
        await client.disconnect()
    """

    def __init__(self):
        self._session: ClientSession | None = None
        self._exit_stack = None

    async def connect(self, server_command: str) -> None:
        """
        连接到 MCP Server。
        server_command: 启动 Server 的命令，如 "python expense_server.py"
        """
        parts = server_command.strip().split()
        command = parts[0]
        args = parts[1:] if len(parts) > 1 else []

        server_params = StdioServerParameters(
            command=command,
            args=args,
            env=None,
        )

        # 使用 asynccontextmanager 管理进程生命周期
        from contextlib import AsyncExitStack
        self._exit_stack = AsyncExitStack()
        
        stdio_transport = await self._exit_stack.enter_async_context(
            stdio_client(server_params)
        )
        read_stream, write_stream = stdio_transport
        
        self._session = await self._exit_stack.enter_async_context(
            ClientSession(read_stream, write_stream)
        )
        
        # 初始化握手（MCP 协议要求）
        await self._session.initialize()
        print(f"[MCP] 已连接到 Server: {server_command.split()[0]}")

    async def disconnect(self) -> None:
        """断开连接，清理进程"""
        if self._exit_stack:
            await self._exit_stack.aclose()
            self._exit_stack = None
            self._session = None

    async def list_tools(self) -> list[dict]:
        """
        获取 MCP Server 提供的工具列表，转换为 OpenAI tools 格式。
        这样 Agent 可以直接把这些工具发给 LLM。
        """
        if not self._session:
            raise RuntimeError("未连接到 MCP Server，请先调用 connect()")

        response = await self._session.list_tools()
        
        openai_tools = []
        for tool in response.tools:
            openai_tools.append({
                "type": "function",
                "function": {
                    "name": tool.name,
                    "description": tool.description,
                    "parameters": tool.inputSchema,
                },
            })
        return openai_tools

    async def call_tool(self, name: str, args: dict) -> str:
        """
        调用 MCP Server 中的工具，返回字符串结果。
        Agent 的工具调用循环直接调用此方法。
        """
        if not self._session:
            raise RuntimeError("未连接到 MCP Server，请先调用 connect()")

        result = await self._session.call_tool(name, args)
        
        # MCP 返回 Content 列表，提取文本
        if result.content:
            texts = [c.text for c in result.content if hasattr(c, "text")]
            return "\n".join(texts)
        return ""

    @asynccontextmanager
    async def connected(self, server_command: str):
        """
        上下文管理器版本，自动管理连接生命周期。
        
        使用示例：
            async with client.connected("python expense_server.py"):
                tools = await client.list_tools()
                result = await client.call_tool("query_expenses", {...})
        """
        await self.connect(server_command)
        try:
            yield self
        finally:
            await self.disconnect()


class MultiMCPClient:
    """
    多 MCP Server 管理器：Agent 可以同时连接多个 MCP Server。
    
    使用示例：
        multi = MultiMCPClient()
        await multi.add_server("expense", "python expense_server.py")
        await multi.add_server("crm", "python crm_server.py")
        
        all_tools = await multi.list_all_tools()  # 所有Server的工具合并
        result = await multi.call_tool("query_expenses", {...})  # 自动路由
    """

    def __init__(self):
        self._clients: dict[str, MCPToolClient] = {}
        self._tool_server_map: dict[str, str] = {}  # tool_name -> server_name

    async def add_server(self, server_name: str, command: str) -> None:
        """添加并连接一个 MCP Server"""
        client = MCPToolClient()
        await client.connect(command)
        self._clients[server_name] = client
        
        # 构建工具路由表
        tools = await client.list_tools()
        for tool in tools:
            tool_name = tool["function"]["name"]
            self._tool_server_map[tool_name] = server_name
        
        print(f"[MultiMCP] 已添加 Server '{server_name}'，工具数: {len(tools)}")

    async def list_all_tools(self) -> list[dict]:
        """获取所有 Server 的工具，合并为一个列表发给 LLM"""
        all_tools = []
        for client in self._clients.values():
            tools = await client.list_tools()
            all_tools.extend(tools)
        return all_tools

    async def call_tool(self, name: str, args: dict) -> str:
        """自动路由到正确的 Server 执行工具"""
        server_name = self._tool_server_map.get(name)
        if not server_name:
            raise KeyError(f"工具 '{name}' 不存在于任何已连接的 Server")
        
        client = self._clients[server_name]
        return await client.call_tool(name, args)

    async def disconnect_all(self) -> None:
        """断开所有 Server 连接"""
        for client in self._clients.values():
            await client.disconnect()
        self._clients.clear()
        self._tool_server_map.clear()
```

```python
        messages.append(message)
        
        if not message.tool_calls:
            return message.content  # 不需要调用工具，直接返回
        
        # 执行所有工具调用（并发）
        tool_results = await asyncio.gather(*[
            multi_client.call_tool(tc.function.name, json.loads(tc.function.arguments))
            for tc in message.tool_calls
        ])
        
        # 把工具结果加入消息历史
        for tc, result in zip(message.tool_calls, tool_results):
            messages.append({
                "role": "tool",
                "tool_call_id": tc.id,
                "content": result,
            })
    
    return "达到最大工具调用次数限制"
```

## 工具的缓存策略

陈浩在测试`query_expenses`的时候，发现同一个问题被问了三次，数据库查询也跑了三次。他早在`ToolRegistry`里设计了`cache_ttl`字段，现在在MCP Server端也需要缓存机制。

```python
# utils/tool_cache.py
"""
工具结果缓存：基于内存的 TTL 缓存，专为 READ 类工具设计。
"""

import json
import time
import hashlib
from typing import Any
from functools import wraps


class ToolResultCache:
    """
    轻量级工具结果缓存。
    
    设计原则：
    - 只缓存幂等操作（相同参数必须返回相同结果）
    - TTL 由工具定义，不是全局统一
    - 缓存键基于工具名 + 参数的哈希值
    """

    def __init__(self):
        self._store: dict[str, tuple[Any, float]] = {}

    def _make_key(self, tool_name: str, args: dict) -> str:
        """生成缓存键（工具名 + 参数的 MD5）"""
        args_str = json.dumps(args, sort_keys=True, ensure_ascii=False)
        content = f"{tool_name}:{args_str}"
        return hashlib.md5(content.encode()).hexdigest()

    def get(self, tool_name: str, args: dict) -> tuple[bool, Any]:
        """
        查询缓存。
        返回 (命中, 结果)，未命中时结果为 None。
        """
        key = self._make_key(tool_name, args)
        if key not in self._store:
            return False, None
        
        value, expire_at = self._store[key]
        if time.time() > expire_at:
            del self._store[key]  # 惰性删除过期缓存
            return False, None
        
        return True, value

    def set(self, tool_name: str, args: dict, value: Any, ttl: int) -> None:
        """写入缓存"""
        key = self._make_key(tool_name, args)
        self._store[key] = (value, time.time() + ttl)

    def invalidate(self, tool_name: str | None = None) -> int:
        """
        主动失效缓存。
        tool_name=None 时清空所有缓存。
        返回删除的条目数。
        """
        if tool_name is None:
            count = len(self._store)
            self._store.clear()
            return count
        
        keys_to_delete = [
            k for k, (v, _) in self._store.items()
            # 缓存键是哈希，无法直接按工具名过滤
            # 实际项目中可以在 key 前缀加工具名
        ]
        return 0

    def stats(self) -> dict:
        """缓存统计信息"""
        now = time.time()
        alive = sum(1 for _, (_, exp) in self._store.items() if now < exp)
        return {
            "total_entries": len(self._store),
            "alive_entries": alive,
            "expired_entries": len(self._store) - alive,
        }


def cached_tool(ttl: int):
    """
    装饰器：为 async 工具函数添加缓存。
    
    用法：
        @cached_tool(ttl=300)
        async def query_expenses(user_id: str, ...) -> list:
            ...
    """
    cache = ToolResultCache()
    
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # 把位置参数和关键字参数统一成字典
            import inspect
            sig = inspect.signature(func)
            bound = sig.bind(*args, **kwargs)
            bound.apply_defaults()
            call_args = dict(bound.arguments)
            
            hit, cached_result = cache.get(func.__name__, call_args)
            if hit:
                return cached_result
            
            result = await func(*args, **kwargs)
            cache.set(func.__name__, call_args, result, ttl)
            return result
        
        wrapper.cache = cache  # 暴露缓存对象，方便手动失效
        return wrapper
    
    return decorator
```

缓存的边界非常清晰：**只有READ类工具的结果可以缓存**。

如果`query_expenses`的结果缓存了5分钟，但用户在这5分钟内提交了新的报销——下一次查询会返回旧数据。这是可接受的（报销记录不是实时需要精确的）。

但`send_message`绝对不能缓存——即使是相同的参数，每次调用都应该发送一条新消息。
